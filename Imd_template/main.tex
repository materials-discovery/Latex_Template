\documentclass{beamer}

% Set theme
\usetheme{default}

% Required packages
\usepackage{graphicx}
\usepackage{tikz} % For adding text on images
\usepackage{fontspec}
\setmainfont{Arial}
\setsansfont{Arial}

% Set the UCL banner at the top with additional text
\setbeamertemplate{background}{
    \begin{picture}(0,0)
        \put(0,-25){% Adjust position of the banner
            \includegraphics[width=\paperwidth]{uclbannerorange.pdf} % Replace with your banner file
        }
        \put(0,-55){% Adjust position of the text
            \begin{tikzpicture}[remember picture, overlay]
                \node[anchor=west, text=white, font=\bfseries\small] at (0, 1.7) {Faculty of Mathematical and Physical Sciences};
                \node[anchor=west, text=white, font=\bfseries\small] at (0, 1.3) {Institute for Materials Discovery};
            \end{tikzpicture}
        }
    \end{picture}
}

% Global color settings: Default to black for all text
\setbeamercolor{normal text}{fg=black} % Default text color
\setbeamercolor{frametitle}{fg=black} % Default frametitle color

\setbeamercolor{section in toc}{fg=black} % Sections in Table of Contents
\setbeamercolor{subsection in toc}{fg=black} % Subsections in Table of Contents
\setbeamercolor{item}{fg=black} % Default list item color

% Frame title customization
\setbeamertemplate{frametitle}{
    \vspace{1cm} % Adjust title position
    \textcolor{black}{\Large\insertframetitle} % Default title font size and color
}

% Reduce margins (optional)
\setbeamersize{text margin left=1.4em, text margin right=1.4em}

% Title, author, and date
\title{\textbf{Title}}
\author{author}
\date{\today}

\begin{document}

% Title page
\frame{\titlepage}

% Table of contents page
\begin{frame}
    \frametitle{\textbf{Table of Contents}}
    \tableofcontents % Automatically generate the table of contents
\end{frame}

% Section 1
\section{1st part}
\begin{frame}
    \frametitle{\textbf{Introduction}} % Default black title
    This is the introduction part
\end{frame}

% Section 2
\section{2nd part}
\begin{frame}
    \frametitle{{\textbf{theory}}} % Blue title for this frame
    This is the content about the Theory.
\end{frame}

% Section 3
\section{3rd}
\begin{frame}
    \frametitle{\textbf{Application}} % Default black title
    \textcolor{red}{This is the content about the applications of it.} % Red text for this frame
\end{frame}

% Section 4
\section{Future}
\begin{frame}
    \frametitle{\textbf{{Future}}} % Green title for this frame
    \textcolor{purple}{
        This section discusses FUTURE.
    } % Purple text for the content
\end{frame}

% Ending page
\begin{frame}{Thank You!}
    Thank you for your attention!
\end{frame}

\end{document}
